function getOrCreateCustomerId() {
  // 1. Check if we already have one saved
  let id = localStorage.getItem('klipy_customer_id');
  
  // 2. If not, create a new one
  if (!id) {
    // Modern browsers have a built-in way to make unique IDs
    id = crypto.randomUUID(); 
    localStorage.setItem('klipy_customer_id', id);
  }
  
  return id;
}

export async function fetchGifs(q, page = 1, per_page = 15) {
  const app_key = KLIPY_KEY; // Ensure this is just the string, no 'Bearer' or 'key='
  const customer_id = getOrCreateCustomerId();
  
  const params = new URLSearchParams({
    page: page.toString(),
    per_page: per_page.toString(),
    q: q,
    customer_id: customer_id,
    locale: "en_US", // Format must be xx_XX
    content_filter: "medium"
  });

  const url = `https://api.klipy.com/api/v1/${app_key}/gifs/search?${params}`;
  
  // Debugging: Copy this URL from your console and paste it into a new tab
  console.log("Klipy Request URL:", url);

  const res = await fetch(url, {
    method: "GET",
    mode: "cors", // Explicitly ask for CORS
  });

  if (res.status === 403) {
    console.error("Access Forbidden: Check if localhost is whitelisted in Klipy Dashboard.");
  }

  if (!res.ok) {
    throw new Error(`Klipy API error: ${res.status}`);
  }

  return await res.json();
}
